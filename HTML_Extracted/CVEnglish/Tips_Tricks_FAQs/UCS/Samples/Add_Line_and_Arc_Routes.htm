Add Line and Arc Routes

It is possible to add Line and Arc Routes through a UCS.  Below is an example of a UCS adding the different types of Route Operations.

;Adds a Line Operation

 

For Each RF Part

 

;Adds Line

Dim LIN as New Line

LIN._FACEWP = 1

LIN.Visible<bool> = True

LIN.DX = 0

LIN.DY = 10

LIN.DZ = 0.25

LIN.X = 5

LIN.Y = 5

LIN.Z = 0

LIN.AX = 0

LIN.AY = 0

LIN.AZ = -45

LIN._RCUT = 2     

LIN.TOOLID = 5

 

;Adds Circle

Dim CIR as New Arc

CIR._FACEWP = 1

CIR.Visible<bool> = True

CIR.DX = 8

CIR.DY = 8

CIR.DZ = 0.25

CIR.X = 15

CIR.Y = 5

CIR.Z = 0

CIR.AX = 0

CIR.AY = 0

CIR.AZ = 0

CIR._RCUT = 2     

CIR.TOOLID = 5

 

;Adds Counter Clockwise Arc (Default)

Dim CCARC as New Arc

CCARC._FACEWP = 1

CCARC.Visible<bool> = False

CCARC.DX = 10

CCARC.DY = 10

CCARC.DZ = 0.25

CCARC.X = 5

CCARC.Y = 15

CCARC.Z = 0

CCARC.AX = 0

CCARC.AY = 0

CCARC.AZ = 0

CCARC._ANG1 = 315

CCARC._ANG2 = 45

CCARC._RCUT = 1     

CCARC.TOOLID = 5

 

;Adds Clockwise Arc

Dim CWARC as New Arc

CWARC._FACEWP = 1

CWARC.Visible<bool> = False

CWARC.DX = 10

CWARC.DY = 10

CWARC.DZ = 0.25

CWARC.X = 15

CWARC.Y = 15

CWARC.Z = 0

CWARC.AX = 0

CWARC.AY = 0

CWARC.AZ = 0

CWARC._ANG1 = 45

CWARC._ANG2 = 315

CWARC._CWARC = 1

CWARC._RCUT = 1     

CWARC.TOOLID = 5
